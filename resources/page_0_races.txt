    jQuery.fn.dataTableExt.oSort['result_place-asc'] = function (a, b) {

        if (!a && !b)
            return 0;

        if (!a)
            return 1;

        if (!b)
            return -1;

        var intValA = parseInt(a);
        var intValB = parseInt(b);

        // A didn't place so B is greater
        if (!intValA && intValB) {
            return 1;
        }

        if (intValA && !intValB) {
            return -1;
        }

        // Both placed
        if (intValA && intValB) {
            return (intValA > intValB ? 1 : -1);
        }

        // Last case, both didn't place
        var orderOfAbnormalResult = ["P", "F", "T", "B", "S", "R"];

        return (jQuery.inArray(orderOfAbnormalResult, a) < jQuery.inArray(orderOfAbnormalResult, b) ? -1 : 1);
    };

    jQuery.fn.dataTableExt.oSort['result_place-desc'] = function (a, b) {
        var ascVal = jQuery.fn.dataTableExt.oSort['result_place-asc'](a, b);
        return ascVal * -1;
    };
